// === ENCODERS ===
bool encoder_update_user(uint8_t index, bool clockwise) {
    if (index == 0) {
        if (clockwise) tap_code(KC_PGDN);
        else tap_code(KC_PGUP);
    } else if (index == 1) {
        uint8_t current_layer = get_highest_layer(layer_state);
        
        if (clockwise) {
            current_layer = (current_layer + 1) % _LAYER_COUNT;
        } else {
            current_layer = (current_layer - 1 + _LAYER_COUNT) % _LAYER_COUNT;
        }
        
        layer_move(current_layer);
    }
    return false;
}

#define NUM_FRAMES 16

static void render_animation(void) {
    if (timer_elapsed(anim_timer) > FRAME_DURATION) {
        anim_timer = timer_read();
        current_frame = (current_frame + 1) % NUM_FRAMES;
    }
    
    // Desenha o frame atual linha por linha (64x64 pixels no lado esquerdo)
    for (uint8_t page = 0; page < 8; page++) {
        for (uint8_t x = 0; x < 64; x++) {
            uint16_t buffer_index = page * 128 + x;
            uint8_t frame_byte = pgm_read_byte(&frame_array[current_frame][page * 64 + x]);
            oled_write_raw_byte(frame_byte, buffer_index);
        }
    }
}

bool oled_task_user(void) {
    // Mostra animao e info de layer
    render_animation();
    
    oled_set_cursor(11, 2);
    oled_write_P(PSTR(">LAYER<"), false);
    
    oled_set_cursor(11, 4);
    switch (get_highest_layer(layer_state)) {
        case _DEFAULT:
            oled_write_P(PSTR("DEFAULT"), false);
            break;
        case _CONTROL:
            oled_write_P(PSTR("CONTROL"), false);
            break;
        case _CENAS:
            oled_write_P(PSTR(" CENAS "), false);
            break;
        default:
            oled_write_P(PSTR("UNKNOWN"), false);
    }
    
    return false;
}
